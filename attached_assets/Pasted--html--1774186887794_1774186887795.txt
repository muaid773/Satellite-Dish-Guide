<script>
    let map = L.map('map').setView([24.7136, 46.6753], 5);
    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
        attribution: '© OpenStreetMap'
    }).addTo(map);

    let marker, line, circle;

    function calculateSatelliteDirection() {
        const lat = parseFloat(document.getElementById('userLat').value);
        const lon = parseFloat(document.getElementById('userLon').value);
        const sLon = parseFloat(document.getElementById('satLon').value);

        if (isNaN(lat) || isNaN(lon) || isNaN(sLon)) {
            alert("يرجى إدخال إحداثيات صحيحة");
            return;
        }

        // تحويل الدرجات إلى راديان
        const rLat = lat * Math.PI / 180;
        const diffLon = (sLon - lon) * Math.PI / 180;

        // 1. حساب السمت (Azimuth)
        let azimuth = 180 + (Math.atan2(Math.sin(diffLon), Math.tan(rLat)) * 180 / Math.PI);
        azimuth = azimuth % 360;

        // 2. حساب الارتفاع (Elevation)
        const R = 6378.137; // نصف قطر الأرض
        const H = 35786 + R; // ارتفاع المدار الثابت + نصف القطر
        const cosG = Math.cos(rLat) * Math.cos(diffLon);
        let elevation = Math.atan2((cosG - (R / H)), Math.sqrt(1 - cosG * cosG)) * 180 / Math.PI;

        // 3. حساب ميل العدسة (Skew)
        let skew = Math.atan2(Math.sin(diffLon), Math.tan(rLat)) * 180 / Math.PI;

        // تحديث الواجهة
        document.getElementById('azimuth').innerText = azimuth.toFixed(2) + "°";
        document.getElementById('elevation').innerText = elevation.toFixed(2) + "°";
        document.getElementById('skew').innerText = skew.toFixed(2) + "°";

        // تحديث الخريطة
        updateMap(lat, lon, azimuth);
    }

    function updateMap(lat, lon, az) {
        if (marker) map.removeLayer(marker);
        if (line) map.removeLayer(line);
        if (circle) map.removeLayer(circle);

        const userPos = [lat, lon];
        map.setView(userPos, 13);

        marker = L.marker(userPos).addTo(map).bindPopup("موقعك").openPopup();

        // رسم دائرة تمثل المحيط
        circle = L.circle(userPos, {
            radius: 500,
            color: 'blue',
            fillOpacity: 0.1
        }).addTo(map);

        // حساب نقطة نهاية الخط باتجاه القمر (طول الخط 1 كم للعرض البصري)
        const distance = 0.01; // مسافة تقريبية للرسم
        const endLat = lat + (distance * Math.cos((90 - az) * Math.PI / 180));
        const endLon = lon + (distance * Math.sin((90 - az) * Math.PI / 180));

        line = L.polyline([userPos, [endLat, endLon]], {
            color: 'red',
            weight: 5,
            opacity: 0.7
        }).addTo(map);
    }
